¿Qué es? Código que se ejecuta ANTES o DESPUÉS de cada petición (como los filtros de Java).


ejemplo en js:

// middlewares/authMiddleware.js - Verificar si el usuario está logueado

function verificarAutenticacion(req, res, next) {
  const token = req.headers['authorization'];
  
  if (!token) {
    return res.status(401).json({ message: 'No autorizado' });
  }
  
  // Verificar token...
  req.usuario = usuarioDecodificado;
  next(); // Continuar con el controlador
}

// middlewares/logger.js - Registrar cada petición
function logger(req, res, next) {
  console.log(`${req.method} ${req.url} - ${new Date().toISOString()}`);
  next();
}

// Uso en app.js
app.use(logger);
app.use(verificarAutenticacion);

¿Qué va aquí?

- Autenticación (verificar tokens)
- Logging (registrar peticiones)
- Validación de datos de entrada
- Compresión de respuestas
- Manejo de errores global
