¿Qué es? El MAPA que conecta cada URL con su controlador.

en js seria:

// routes/tratamientoRoutes.js
const express = require('express');
const router = express.Router();
const tratamientoController = require('../controllers/tratamientoController');

// Definir las rutas
router.get('/', tratamientoController.listar);           // GET /api/tratamientos
router.get('/top3', tratamientoController.top3);         // GET /api/tratamientos/top3
router.get('/:id', tratamientoController.obtenerPorId);  // GET /api/tratamientos/5
router.post('/', tratamientoController.crear);           // POST /api/tratamientos
router.put('/:id', tratamientoController.actualizar);    // PUT /api/tratamientos/5
router.delete('/:id', tratamientoController.eliminar);   // DELETE /api/tratamientos/5

module.exports = router;

¿Qué va aquí?

- SOLO definiciones de URLs
- NO lógica de negocio
- NO consultas a BD
